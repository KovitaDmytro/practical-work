Вступ

HTML (HyperText Markup Language) є основною мовою для створення веб-сторінок. Вона визначає структуру та семантику вмісту веб-документа, що дозволяє браузерам правильно відображати інформацію для користувача. HTML є однією з основних складових веб-технологій і є фундаментальним елементом для розробки веб-сторінок та веб-додатків.

Історія HTML

HTML була розроблена Тімоті Бернерсом-Лі у 1991 році в рамках проекту World Wide Web (WWW) у Європейському організації ядерних досліджень (CERN). Початково HTML була простою мовою для створення гіпертекстових документів, яка складалася з набору тегів, що визначали структуру тексту та посилання між сторінками.

Розвиток HTML

Протягом років HTML зазнавала численних змін і вдосконалень. Послідовні версії мови додавали нові функції, елементи та можливості, що розширювали її функціональність. Однією з ключових подій у розвитку HTML було введення стандарту HTML5 у 2014 році, який приніс значні зміни і нові можливості для веб-розробників.

Роль HTML у веб-розробці

HTML відіграє важливу роль у веб-розробці, визначаючи структуру та семантику веб-сторінок. Вона дозволяє розробникам створювати доступні та ефективні веб-додатки, які можуть взаємодіяти з користувачами через різноманітні пристрої та платформи.

Теґ <ul>

Маркований (невпорядкований) список в HTML визначається тегом <ul> (Unordered List Element). Кожен елемент списку повинен починатися з тега <li> (скорочене від англійського list item - елемент списку).


   <ul>


   <li>елемент позначеного списку</li>


   </ul>

Теґ <ol>

Тег <ol> (агнл. ordered (enumerated) list - визначає впорядкований (пронумерований) список.


   <ol>


   <li>елемент позначеного списку</li>


   </ol>

Теґ <li>

Тег <li> (англ. list item - елемент списку) створює елемент впорядкованого (нумерованого) або маркованого (ненумерованого) списку. Тег <li> використовується у впорядкованих списках (<ol>), марковані списки (<ul>).


   <ul>

   <li>елемент маркорованого списку</li>

   </ul>


   <ol>

   <li>елемент нумерованого списку</li>

   </ol>

Теґ <dl>

HTML тег <dl> (англ. definition list - список визначень) призначених для створення списку термінів (списку описів/визначень).

Теґ <dt>

Тег <dt> входить в трійку елементів <dl>, <dt>, <dd>, призначених для створення списку визначень. Кожен такий список починається з контейнера <dl>, куди входить тег <dt> створює термін і тег <dd> задає визначення цього терміна. Закриває тег </dt> не обов'язковий, оскільки наступний тег повідомляє про завершення попереднього елемента. Проте, хорошим стилем є закривати всі теги.

Теґ <dd>

HTML тег <dd> (англ. define data - визначити дані). Цей тег задає визначення до терміну, котрий визначений елементом <dt>


   <dl>

    <dt>Термін</dt>

    <dd>Опис терміна</dd>

   </dl>

Теґ <table>

Теги <table>, <thead>, <tbody>, <tr>, <th>, <td></h3> використовуються для створення табличних структур у HTML. Ось коротка інформація про кожен з цих тегів:

<table>: Визначає таблицю.

<thead>: Визначає заголовок таблиці.

<tbody>: Визначає тіло таблиці.

<tr>: Визначає рядок у таблиці.

<th>: Визначає заголовок стовпця. Зазвичай використовується для першого рядка або стовпця у таблиці.

<td>: Визначає дані у таблиці. Зазвичай використовується для решти рядків та стовпців таблиці.


   <p>

       <thead>

           <tr>

                <td> ... </td>

           </tr>

       </thead>

       <tbody> ... </tbody>

   </p>

.var table = document.getElementById("editableTable");

    function addRow() {
        var row = table.insertRow(-1);
        var cell1 = row.insertCell(0);
        var cell2 = row.insertCell(1);
        var cell3 = row.insertCell(2);

        cell1.contentEditable = true;
        cell2.contentEditable = true;
        cell3.contentEditable = true;
    }

    function editRow(button) {
        var row = button.parentNode.parentNode;
        var cells = row.getElementsByTagName("td");
        for (var i = 0; i < cells.length - 1; i++) {
            cells[i].setAttribute("contenteditable", "true");
        }
        button.style.display = "none";
        button.parentNode.querySelector(".saveBtn").style.display = "inline-block";
    }

    function saveRow(button) {
        var row = button.parentNode.parentNode;
        var cells = row.getElementsByTagName("td");
        for (var i = 0; i < cells.length - 1; i++) {
            cells[i].setAttribute("contenteditable", "false");
        }
        button.style.display = "none";
        button.parentNode.querySelector(".editBtn").style.display = "inline-block";
    }

    function saveTable() {
        var data = [];

                rowData.push(cells[j].innerText);
            }
            data.push(rowData);

JavaScript 

function sayHello() {
    let name = document.getElementById("name").value;
    alert(`Привіт, ${name}!`);
}

function calculateSum() {
    let num1 = parseFloat(document.getElementById("num1").value);
    let num2 = parseFloat(document.getElementById("num2").value);
    let sum = num1 + num2;
    document.getElementById("sumResult").innerHTML = `Сума: ${sum}`;
}

function generateRandom() {
    let min = parseInt(document.getElementById("min").value);
    let max = parseInt(document.getElementById("max").value);
    let randomNum = Math.floor(Math.random() * (max - min + 1)) + min;
    document.getElementById("randomResult").innerHTML = `Випадкове число: ${randomNum}`;
}

function checkPasswordStrength() {
    let password = document.getElementById("password").value;
    let strength = "";
    if (password.length < 6) {
        strength = "Дуже слабкий";
    } else if (password.length < 10) {
        strength = "Слабкий";
    } else if (password.length < 14) {
        strength = "Достатній";
    } else {
        strength = "Сильний";
    }
    document.getElementById("passwordStrength").innerHTML = `Сила пароля: ${strength}`;
}

function calculateFactorial() {
    let num = parseInt(document.getElementById("number").value);
    let factorial = 1;
    for (let i = 1; i <= num; i++) {
        factorial *= i;
    }
    document.getElementById("factorialResult").innerHTML = `Факторіал ${num}: ${factorial}`;
}

function checkPrime() {
    let num = parseInt(document.getElementById("prime").value);
    let isPrime = true;
    if (num === 1) {
        isPrime = false;
    } else {
        for (let i = 2; i <= Math.sqrt(num); i++) {
            if (num % i === 0) {
                isPrime = false;
                break;
            }
        }
    }
    if (isPrime) {
        document.getElementById("primeResult").innerHTML = `${num} - просте число`;
    } else {
        document.getElementById("primeResult").innerHTML = `${num} - не просте число`;
    }
}

function reverseString() {
    let str = document.getElementById("inputString").value;
    let reversed = str.split('').reverse().join('');
    document.getElementById("reversedString").innerHTML = `Реверс рядка: ${reversed}`;
}

function checkPalindrome() {
    let word = document.getElementById("palindrome").value;
    let reversedWord = word.split('').reverse().join('');
    if (word.toLowerCase() === reversedWord.toLowerCase()) {
        document.getElementById("palindromeResult").innerHTML = `Це паліндром!`;
    } else {
        document.getElementById("palindromeResult").innerHTML = `Це не паліндром!`;
    }
}

function findMax() {
    let numbers = document.getElementById("numberList").value.split(',').map(Number);
    let max = Math.max(...numbers);
    document.getElementById("maxNumber").innerHTML = `Найбільше число: ${max}`;
}

JavaScript 

    let marqueeElement = document.querySelector('.animation-language');

    let texts = ['JavaScript', 'Python', 'Java', 'C++',"TypeScript"];

    marqueeElement.style.animationPlayState = 'running';

    let delay = 500;

    function controlMarquee() {
        setTimeout(function() {
            marqueeElement.style.animationPlayState = 'paused';
            let randomIndex = Math.floor(Math.random() * texts.length);
            marqueeElement.innerText = texts[randomIndex];
            setTimeout(function() {
                marqueeElement.style.animationPlayState = 'running';
                controlMarquee();
            }, delay);
        }, delay);
    }
    controlMarquee();
